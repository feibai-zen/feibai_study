54   class Folder implements AbstractFile {
55   	private String name;
56   	// 定义容器，用来存放本容器构建下的子节点
57   	private List<AbstractFile> list = new ArrayList<AbstractFile>();
58   
59   	public Folder(String name) {
60   		super();
61   		this.name = name;
62   	}
63   
64   	public void add(AbstractFile file) {
65   		list.add(file);
66   	}
67   
68   	public void remove(AbstractFile file) {
69   		list.remove(file);
70   	}
71   
72   	public AbstractFile getChild(int index) {
73   		return list.get(index);
74   	}
75   
76   	@Override
77   	public void killVirus() {
78   		System.out.println("---文件夹：" + name + ",进行查杀");
79   
80   		for (AbstractFile file : list) {
81   			file.killVirus();
82   		}
83   	}
84   
85   }
