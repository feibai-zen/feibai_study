3    import java.util.ArrayList;

12   public class ConcreteMyAggregate {
13   	private List<Object> list = new ArrayList<Object>();
14   
15   	public void addObject(Object obj) {
16   		this.list.add(obj);
17   	}
18   
19   	public void removeObject(Object obj) {
20   		this.list.remove(obj);
21   	}
22   
23   	public List<Object> getList() {
24   		return list;
25   	}
26   
27   	public void setList(List<Object> list) {
28   		this.list = list;
29   	}
30   
31   	// 获得迭代器
32   	public MyIterator createIterator() {
33   		return new ConcreteIterator();
34   	}
35   
36   	// 使用内部类定义迭代器，可以直接使用外部类的属性
37   	private class ConcreteIterator implements MyIterator {
38   
39   		private int cursor; // 定义游标用于记录遍历时的位置
40   
41   		@Override
42   		public void first() {
43   			cursor = 0;
44   		}
45   
46   		@Override
47   		public Object getCurrentObj() {
48   			return list.get(cursor);
49   		}
50   
51   		@Override
52   		public boolean hasNext() {
53   			if (cursor < list.size()) {
54   				return true;
55   			}
56   			return false;
57   		}
58   
59   		@Override
60   		public boolean isFirst() {
61   			return cursor == 0 ? true : false;
62   		}
63   
64   		@Override
65   		public boolean isLast() {
66   			return cursor == (list.size() - 1) ? true : false;
67   		}
68   
69   		@Override
70   		public void next() {
71   			if (cursor < list.size()) {
72   				cursor++;
73   			}
74   		}
75   
76   	}
77   
78   }
