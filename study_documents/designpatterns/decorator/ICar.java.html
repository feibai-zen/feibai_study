3    /**
4     * 抽象构建
5     * 
6     * @author Administrator
7     *
8     */
9    public interface ICar {
10   	void move();
11   }
12   
13   //ConcreteComponent 具体构件角色(真实对象)
14   class Car implements ICar {
15   	@Override
16   	public void move() {
17   		System.out.println("陆地上跑！");
18   	}
19   }
20   
21   //Decorator装饰角色
22   class SuperCar implements ICar {
23   	protected ICar car;
24   
25   	public SuperCar(ICar car) {
26   		super();
27   		this.car = car;
28   	}
29   
30   	@Override
31   	public void move() {
32   		car.move();
33   	}
34   }
35   
36   //ConcreteDecorator具体装饰角色
37   class FlyCar extends SuperCar {
38   
39   	public FlyCar(ICar car) {
40   		super(car);
41   	}
42   
43   	public void fly() {
44   		System.out.println("天上飞！");
45   	}
46   
47   	@Override
48   	public void move() {
49   		super.move();
50   		fly();
51   	}
52   
53   }
54
